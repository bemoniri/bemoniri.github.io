	\section{Introduction}
	\frame{\sectionpage}
		
	\begin{frame}
		\frametitle{Non-Parametric Least Squares}
		\begin{itemize}
			\item \textbf{MMSE Estimation}:
			\begin{equation*}
			\bar\cL_f = \E_{X, Y} [(Y - f\,(X))^2] \implies \fstar(x) = \E[Y\,|X=x\,]
			\end{equation*}
			
			\item In practice  we are given a collection of samples $\{(\bx_i, y_i)\}_{i = 1}^n$, which can be used to compute an empirical analog of the MSE:			
			\begin{equation*}
				\widehat\cL_f = \frac{1}{n} \sum_{i = 1}^{n} \big(y_i - f\,(x_i\,)\big)^2
			\end{equation*}
			
			\item \textbf{Non-Parametric Least Squares}: Minimizing this  criterion over some suitably controlled function class.
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Different Measures of Quality}
		\begin{itemize}
			\item \textbf{Excess Risk}:
			\begin{equation*}
				||f-\fstar||_{L^2(\Prob)}= \E_{X\sim \Prob}\Big[\big(f\,(X\,) - \fstar(X\,)\big)^2\Big]
			\end{equation*}
			where $\Prob$ denotes the distribution over covariates.
			
			\item \textbf{This Chapter}: Let $\{\bx_i\}_{i = 1}^{n}$ be the set of fixed covariates and $\Prob_n$ be their Empirical measure. Define:
			\begin{equation*}
				||f-\fstar||_{L^2(\Prob_n)} = \Bigg[\frac{1}{n}\sum_{i = 1}^n \big(f\,(\bx_i) - \fstar(\bx_i)\big)^2\Bigg]^{1/2}.
			\end{equation*}
			In this talks, we will use this measure and denote it as $||f-\fstar||_{n}$.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Fixed or Stochastic Covariates?}
		\begin{block}{Note}
			\begin{itemize}
				\item We will view the samples $\{\bx_i\}_{i = 1}^n$
				as being fixed, a set-up known as \textit{regression with a fixed design}.
				\item Results from Chapter 14 to follow can be used to translate these bounds into equivalent results in the population $L_2(\Prob)$-norm.
			\end{itemize} 
		\end{block}		
	\end{frame}

	\begin{frame}
	\frametitle{Estimation via Least Squares}	
		Given a fixed collection $\{\bx_i\}_{i = 1}^n$, model the responses as 		
		\begin{equation*}
			y_i = \fstar (\bx_i) + \nu_i, \;\;\; \text{for } i = 1, 2, \dots, n.
		\end{equation*}
		where $\nu_i = \sigma w_i$ in which  $w_i \sim \mathcal{N}(0, 1)$. The least squares estimate is given by the function
		\begin{equation*}
		\fhat \in \arg\min_{f\,\in\cF} \Bigg\{\frac{1}{n} \sum_{i = 1}^{n} \big(y_i - f\,(x_i\,)\big)^2\Bigg\}.
		\end{equation*}
		
		\begin{itemize}
			\item When $\nu_i \sim \mathcal{N}(0,\sigma^2)$, the LS estimate is equivalent to the constrained maximum likelihood.
			
			\item When $\cF$ is an RKHS, it can also be convenient to use regularized estimators of the form:			
			\begin{equation*}
			\fhat \in \arg\min_{f\,\in\cF} \Bigg\{\frac{1}{n} \sum_{i = 1}^{n} \big(y_i - f\,(x_i\,)\big)^2 + \lambda_n ||f\,||_\cF^2\Bigg\}.
			\end{equation*}
		\end{itemize}
	\end{frame}

	\section{Examples}
	\frame{\sectionpage}
		\begin{frame}
		\frametitle{Parametric: Linear Regression}
		
		\begin{itemize}
			\item For a given vector $\boldsymbol{\theta} \in \mathbb{R}^d$, define  $f_{\boldsymbol{\theta}}(\bx) = \langle \boldsymbol{\theta}, \bx \rangle$ and consider the function class
			$\cF_C = \{f_{\boldsymbol{\theta}}: \mathbb{R}^d \to \mathbb{R} | \boldsymbol{\theta} \in C\}$
			for a compact $C$.
			\item The least squares estimate:
			\begin{equation*}
			\widehat{\boldsymbol{\theta}} \in \arg\min_{\boldsymbol{\theta} \in C} \Bigg\{\frac{1}{n} ||y - \mathbf{X}\boldsymbol{\theta}||_2^2\Bigg\}
			\end{equation*}
			where $\mathbf{X} \in \mathbb{R}^{n\times d}$ is the design matrix.
		\end{itemize}
		
		 
		\begin{itemize}
			\item The constrained form of Ridge Regression:
			\begin{equation*}
				C = \big\{\boldsymbol{\theta}\in \mathbb{R}^d \;\big|\; ||\boldsymbol{\theta}||_2^2\leq R\big\}.
			\end{equation*}
			
			\item The constrained form of LASSO:
			\begin{equation*}
			C = \big\{\boldsymbol{\boldsymbol{\theta}} \in \mathbb{R}^d \;\big|\; ||\boldsymbol{\theta}||_1\leq R\big\}.
			\end{equation*}
		\end{itemize}		
		\end{frame}
	
		\begin{frame}

		\frametitle{Non-Parametric: Cubic Smoothing Spline}
		Consider the class of twice cont. differentiable functions $f: [0, 1] \to \mathbb{R}$ and for a given radius $R$, define:
		\begin{equation*}
		\cF(R) = \Bigg\{f: [0, 1] \rightarrow \mathbb{R} \;\Bigg|\; \int_{0}^{1} \Big(f\;''(x)\Big)^2 dx \leq R\Bigg\}.
		\end{equation*}
		The integral can be understood as a Hilbert norm bound in the second-order Sobolev space.
		\end{frame}
	
	
		\section{Critical Radius: Bounding the Prediction Error}
		\frame{\sectionpage}
		
		\begin{frame}
			\frametitle{Critical Radius: Bounding the Prediction Error}
			\begin{itemize}

			\item Intuitively, the difficulty of estimating the function $\fstar$ should depend on the complexity of the function class $\cF$ in which it lies.
			\item  \textbf{a localized form of Gaussian complexity:} it measures the complexity of the function class $\cF$, locally in a neighborhood around the true regression function $\fstar$.
			\end{itemize}		
			\begin{block}{Local Gaussian Complexity}
				Define $\c\fstar = \cF - \{\fstar\} = \{f - \fstar | f \in \cF\}$. For a given radius $\delta > 0$, \textit{the local Gaussian complexity} around $\fstar$ at scale $\delta$ is given by:
				\begin{equation*}
					\cG_n(\delta; \c\fstar) = \E_w \Bigg[\sup_{g \in \cF^\star,\, ||g||_n \leq \delta} \Big|\frac{1}{n} \sum_{i = 1}^{n} w_i\, g\,(\bx_i)\Big|\Bigg]
				\end{equation*}				
			\end{block}		
		\end{frame}
	
				\begin{frame}
			\frametitle{Critical Inequality}
			
			\begin{block}{Critical Inequality}
				A central object in our analysis is the set of positive scalars $\delta$ that satisfy the \textit{critical inequality}:
				\begin{equation*}
				\frac{\delta}{2\sigma} \geq \frac{\cG_n(\delta; \cF^\star)}{\delta}
				\end{equation*}			
			\end{block}
			
			\begin{itemize}							
				\item We refer to any $\delta > 0$ satisfying the critical inequality as being valid.
				\item We use $\delta^\star_n$ to denote the smallest positive radius for which the critical inequality holds.
			\end{itemize}
		\end{frame}
		
		\begin{frame}
			\frametitle{Some Intuition}
			\begin{itemize}
				\item Note that $\frac{1}{2n} \sum_{i = 1}^{n} \big(y_i - \widehat f\, (\bx_i)\big)^2 \leq \frac{1}{2n}\sum_{i = 1}^{n} \big(y_i - \fstar(\bx_i)\big)^2$.
				\vspace{0.2cm}
				
				\item 					
				After some computations: $\frac{1}{2} ||\fhat - \fstar||_n^2 \leq \frac{\sigma}{n} \sum_{i = 1}^{n} w_i\,\big(\fhat\,(\bx_i) - \fstar(\bx_i)\big)$.
				\vspace{0.2cm}
				
				\item  Note that $\fhat - \fstar \in \c\fstar$.  Reasoning hueristically, this observation suggests that 				
				\begin{equation*}
					\frac{\delta^2}{2} \leq \sigma \cG_n(\delta; \mathcal{F}^\star) \text{ or equivalently } \frac{\delta}{2\sigma} \leq \frac{\cG_n(\delta; \cF^\star)}{\delta}.
				\end{equation*}	
				
				\item We will present a rigorous result later in this talk.
			\end{itemize}			 
		\end{frame}
		
			
		
		\begin{frame}
			\frametitle{The Main Result}
			
				\begin{block}{Star-Shaped Classes}
				a function class $\cF$ is star-shaped if for any $\alpha \in [0,1]$ we have
				\begin{equation*}
				f \in \cF \implies \alpha f \in \cF.
				\end{equation*}
			\end{block}
			
			\begin{block}{Theorem}
				Suppose that the shifted function class $\cF^\star$ is star-shaped, and let $\delta$ be any solution to the critical inequality. Then for any $t \geq \delta$, the nonparametric least-squares estimate $\widehat{f_n}$ satisfies the bound
				\begin{equation}
					\Prob\big[||\widehat{f_n} - \fstar ||_n^2 \geq 16t\delta\big] \leq \exp\Big(\frac{-nt\delta}{2\sigma^2}\Big)
				\end{equation}
			\end{block}
		\end{frame}

		\begin{frame}
			\frametitle{Properties of Star-Shaped Classes}
			\begin{itemize}
				\item The star shaped condition on $\cF^\star$ is needed in various parts of the proof, including ensuring a valid radii $\delta$ exists.
				
				\item $\cF^\star$ Star shaped = $\cF$ is star shaped around $\fstar$.				
				
				\begin{block}{Property}
					If $\cF$ is convex, it is star-shaped around any $\fstar$. 				
				\end{block}
				
				\begin{block}{Proof}
					Let $\widetilde{f} = f - \fstar \in \cF^\star$ be an arbitrary point in $\cF^\star$ with $f, \fstar \in \cF$.\\
					For any $\alpha \in [0, 1]$, by convexity we have $g = \alpha f + (1-\alpha) \fstar \in \cF$.\\			
					Hence $\alpha \ftilde = \alpha(f - \fstar) = g - \fstar \in \cF^\star$, proving $\cF^\star$ is star-shaped.			
				\end{block}
				\begin{block}{Conversely}
					Similarly if $\cF$ is not convex, there must exists a $\fstar$ such that $\cF$ is not star shaped around $\fstar$.	
				\end{block}
			\end{itemize}						
		\end{frame}
	
			
	
	\begin{frame}
		\frametitle{Properties of Star-Shaped Classes}
		\begin{itemize}
			\item If the star-shaped condition fails to hold, then the main Theorem can instead by applied with $\delta$ defined in terms of the star hull (we will see next session.)
			\begin{equation*}
				\mathrm{star}(\cF^\star) = \big\{\alpha(f-\fstar) | f \in \cF, \alpha \in [0, 1]\big\}.
			\end{equation*}			
			
			\begin{block}{Existence of Critical Radius}
				For any star shaped class $\mathcal{F}^\star$, the function $\delta \rightarrow \frac{\cG_n(\delta, \cF^\star)}{\delta}$ is non-increasing on $(0, \infty)$. Consequently, for any $c>0$, the inequality 
				\begin{equation*}
			\frac{\cG_n(\delta, \cF^\star)}{\delta} \leq c\delta
			\end{equation*}
				has a smallest positive solution.
			\end{block}
		\end{itemize}			
	\end{frame}
		\begin{frame}
		\frametitle{Proof of the Existence of Critical Radius}
			Given $0< \delta \leq t$, we should show that $\frac{\delta}{t}\cG_n(t) \leq \cG_n(\delta)$. 
			
			Given any $h \in \cF^\star$ with $||h||_n\leq t$, define
			$\widetilde{h} = \frac{\delta}{t}h \in \cF^\star$ (by star-shaped assumption) and write
			\begin{equation*}
				\frac{1}{n}\Big\{\frac{\delta}{t} \sum_{i = 1}^{n} w_i h(\bx_i)\} = \frac{1}{n}\Big\{ \sum_{i = 1}^{n} w_i \widetilde{h}(\bx_i)\}.
			\end{equation*}
			By construction, $||\widetilde{h}||_n\leq \delta$. Consequently:
			
			\begin{itemize}
				\item  The RHS is at most $\cG_n(\delta)$ in expectation.
				\item Taking supremum  over the set $\cF \cap \{||h||_n\leq t\}$ following by expectation, yields $\frac{\delta}{t}\cG_n(t)$ on the left hand side. 
			\end{itemize}
		This concludes the proof. \qed
			 
		\end{frame}

		\begin{frame}
			\frametitle{Proof of the main theorem}
			\begin{block}{Theorem}
				Suppose that the shifted function class $\cF^\star$ is star-shaped, and let $\delta$ be any solution to the critical inequality. Then for any $t \geq \delta$, the nonparametric least-squares estimate $\widehat{f_n}$ satisfies the bound
				\begin{equation*}
				\Prob\big[||\widehat{f_n} - \fstar ||_n^2 \geq 16t\delta\big] \leq \exp\Big(\frac{-nt\delta}{2\sigma^2}\Big)
				\end{equation*}
			\end{block}
		\end{frame}
	
	
		\begin{frame}
		\begin{itemize}
		\item Denote $\widehat\Delta = \; \fhat - \fstar \in \cF^\star$, we have $\frac{1}{2}||\widehat\Delta||_n^2 \leq \frac{\sigma}{n}\sum_{i = 1}^{n}w_i \widehat\Delta (\bx_i)$.  We need to control the stochastic component of the right-hand side.
		
		\begin{block}{Auxiliary Lemma}
		Let $\cF^\star$ be a star-shaped class and $\delta$ satisfy the critical inequality.  Define the bad event $A(u)$ as
		\begin{small}
		
		\vspace{-0.3cm}
		\begin{equation*}
		A(u) = \Bigg\{\exists g \in \cF^\star, ||g||_n \geq u \;:\; \Big|\frac{\sigma}{n}\sum_{i = 1}^{n} w_i g(\bx_i)\Big|\geq 2u ||g||_n\Bigg\}.
		\end{equation*}
		\end{small}
		\vspace{-0.3cm}
		
		For a given $u \geq \delta$, we have 
		$\Prob[A(u)]\leq \exp\big(\frac{-nu^2}{2\sigma^2}\big)$.
		\end{block}
		 Now consider two cases:
		\begin{itemize}
			\item \textbf{Case 1:} $||\widehat\Delta||_n < \sqrt{t\delta}$ which implies $||\widehat\Delta||_n^2\leq t\delta$ trivially.
			\item \textbf{Case 2:} $||\widehat\Delta||_n \geq \sqrt{t\delta}$.  We condition on $A^c(\sqrt{t\delta})$. 		Set $u = \sqrt{t\,\delta}$, so that we have $\Prob[A^c(\sqrt{t\,\delta})]\geq 1 - \exp(\frac{-nt\delta}{2\sigma^2})$. Hence:
			\begin{equation*}
				||\widehat\Delta||_n^2 \leq 2\Big|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i \widehat\Delta (\bx_i)\Big| \leq 4||\widehat\Delta||_n\sqrt{t\delta} \implies  ||\Delta||_n^2 \leq 16t\delta.\qed
			\end{equation*}
		\end{itemize}
	\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Proof of the Auxiliary Lemma}
		Recall 
		$A(u) = \Big\{\exists g \in \cF^\star, ||g||_n \geq u \;:\; \Big|\frac{\sigma}{n}\sum_{i = 1}^{n} w_i g(\bx_i)\Big|\geq 2u ||g||_n\Big\}.$\\
		\textbf{Step 1}: reduce  to controlling a $\sup$ over a subset with $||\widetilde g||_n\leq u$.
		\begin{itemize}		
			\item Suppose that $A(u)$ is true, hence  there exists				
			$g \in \cF^\star$ with $||g||_n\geq u$ such that			 
			 \begin{small}
			 \begin{equation*}
			 	\Big|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i g(\bx_i)\Big|\geq 2||g||_n u.
			 \end{equation*}
			 \end{small}
			\item Define $\widetilde{g} = \frac{u}{||g||_n}g \in \cF^\star$, we observe that $||\widetilde{g}||_n = u$. We have 
			\begin{small}
			\begin{equation*}
				\Big|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i\,\widetilde{g}\,(\bx_i)\Big| = \frac{u}{||g||_n}\Big|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i\,g\,(\bx_i)\Big|\geq 2u^2.
			\end{equation*}
			\end{small}
			\item We thus conclude that:
			\begin{small}
				\begin{equation*}
				\Prob[A(u)] \leq \Prob[Z_n(u)\geq 2u^2], \; Z_n(u) := \sup_{\widetilde{g}\,\in \cF^\star, \; ||\,\widetilde{g}\,||_n\, \leq u}\bigg|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i\,\widetilde{g}(\bx_i)\bigg|.
				\end{equation*}
			\end{small}			
		\end{itemize}		
	\end{frame}

	\begin{frame}
		\frametitle{Proof of the Auxiliary Lemma}
		\begin{small}
			$\Prob[A(u)] \leq \Prob[Z_n(u)\geq 2u^2], \; Z_n(u) := \sup_{\widetilde{g}\,\in \cF^\star, \;\;\; ||\,\widetilde{g}\,||_n\, \leq u}\bigg|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i\,\widetilde{g}\,(\bx_i)\bigg|.$
		\end{small}
		
		\textbf{Step 2}: Concentration of suprema.
		\begin{itemize}			

			\item Recall that $w_i \sim \mathcal{N}(0, 1)$, hence
			$\frac{\sigma}{n}\sum_{i = 1}^{n}w_i \,\widetilde{g}\,(\bx_i) \sim \mathcal{N}\Big(0, \frac{\sigma^2}{n}||\widetilde{g}\,||_n^2\Big). $
				
			\item
			$Z_n(u) = h(w_1, \dots, w_n)  := \sup_{\;\widetilde{g}\,\in \cF^\star, \;\;\; ||\,\widetilde{g}\,||_n\, \leq u}\bigg|\frac{\sigma}{n}\sum_{i = 1}^{n}w_i\,\widetilde{g}\,(\bx_i)\bigg|.$
			
			\item
			The Lipschitz constant of $h$ is at most $\frac{\sigma u}{\sqrt{n}}$.
			
			\item We know that for $Z_i \sim \mathcal{N}(0, \sigma^2)$ and $h$ a L-Lipschitz function:
			\begin{small}
				\begin{equation*}
				\Prob[h(Z\,) - \E h(Z\,) \geq t]\leq e^{-\frac{s^2}{2L^2\sigma^2}}.
				\end{equation*}
			\end{small}
			\vspace{-0.5cm}
			\item Let $s = u^2$, we obtain $\Prob(Z_n(u) \geq \E[Z_n(u)\,] + u^2) \leq \exp\big(\frac{-nu^2}{2\sigma^2}\big)$.

			
		\end{itemize}		
	\end{frame}


	\begin{frame}
		\frametitle{Proof of the Auxiliary Lemma}
		\textbf{Step 3}: Bounding the expectation.
		\begin{itemize}					
		\item Note that $\E[Z_n(u)] = \sigma\, \cG_n(u)$.
		\item We know that $\nu \to \frac{\cG_n(\nu)}{\nu}$ is non-increasing. We have also assumed that $u\geq \delta$.
		\begin{equation*}
		\sigma\frac{\cG_n(u)}{u}\leq\sigma\frac{\cG_n(\delta)}{\delta} \leq \frac{\delta}{2}\leq \delta \implies \E[Z_n(u)]\leq u\delta \leq u^2.
		\end{equation*}
		Comibining our results, we obtain 
		\begin{equation*}
			\Prob[A(u)] \leq \Prob[Z_n(u) \geq 2u^2] \leq \Prob(Z_n(u) \geq \E[Z_n(u)] + u^2) \leq \exp\Big(\frac{-nu^2}{2\sigma^2}\Big),
		\end{equation*}
		Which concludes the proof.\qed
	\end{itemize}
	\end{frame}

	\section{How to Compute the Critical Radius?}
	\frame{\sectionpage}
	
	\begin{frame}
		\frametitle{Bounds via Metric Entropy}
		For any star-shaped function class $\cF^\star$, define:
		\begin{itemize}
			\item $B\,(\delta\,; \cF^\star) = \{h\in \cF^\star \;\big|\; ||h||_n\leq \delta\}$
			\item $\mathcal{N}\,\big(t; B_n(\delta; \cF^\star)\big)$ be the $t$-convering number of $B_n(\delta; \cF)$ in norm $||.||_n$.
		\end{itemize}

		\begin{block}{Critical Inequality via Metric Entropy}
			Any $\delta \in (0, \infty]$ satisfying
			\begin{equation*}
				\frac{16}{\sqrt{n}} \int_{\frac{\delta^2}{4\sigma}}^{\delta} \sqrt{\log \mathcal{N}\big(t\,; B(\delta; \cF^\star)\big)} \;dt \leq \frac{\delta^2}{4\sigma},
			\end{equation*}
			satisfies the critical inequality.
		\end{block}
	\end{frame}


	\begin{frame}
		\frametitle{Bounds via Metric Entropy: Proof}
		\begin{itemize}
		\item		
		Construct a $\frac{\delta^2}{4\sigma}$-covering of the set $B\,(\delta; \cF)$, say $\{g\,^1, \dots, g\,^M\}$. For any function $g\in B(\delta; \cF)$, there is a $j \in [M]$ such that $||g\,^j - g||_n \leq \frac{\delta^2}{4\sigma}$. 
		\item  We have
		\begin{small}
		\begin{align*}
			\Big|\frac{1}{n}\sum_{i = 1}^{n} &w_i\, g\,(\bx_i) \Big| \leq \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, g\,^j(\bx_i) \Big| + \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, \big(g\,(\bx_i) - g\,^j(\bx_i)\big) \Big|\\
			&\leq  \max_{j = 1, \dots, M} \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, g\,^j(\bx_i) \Big| + \sqrt{\frac{\sum_{i = 1}^{n}w_i\,^2}{n}} \sqrt{\frac{\sum_{i = 1}^{n}(g\,(\bx_i) - g\,^k(\bx_i))^2}{n}}\\
			&\leq  \max_{j = 1, \dots, M} \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, g\,^j(\bx_i) \Big| + \frac{\delta^2}{4\sigma}\sqrt{\frac{\sum_{i = 1}^{n}w_i\,^2}{n}}.
		\end{align*}
		Hence 
		\begin{equation*}
			\cG_n(\delta) \leq \E\Bigg[\max_{j = 1, \dots, M} \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, g\,^j(\bx_i) \Big|\Bigg] + \frac{\delta^2}{4\sigma}.
		\end{equation*}
				\end{small}
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Bounds via Metric Entropy: Proof}
		\begin{itemize}	
		\item
		To upper bound the first term, we use Dudley's Integral. Define $Z(g\,^j) = \frac{1}{\sqrt{n}}\sum_{i = 1}^{n}w_i\, g\,^j\,(\bx_i)$ for $j = 1, \dots, M$. 
		\item The induced metric: 
		\begin{small}
			$\rho_Z^2\,(g\,^j, g\,^k) := \mathrm{Var}\big(Z(g\,^j) - Z(g\,^k)\big) = ||g\,^j - g\^k||_n^2$
		\end{small}

		\item
		Since $||g\,||_n \leq \delta$ for all $g \in B_n(\delta, \cF\,^*)$, the coarsest resolution of the chaining can be set to $\delta$. We can terminate it at $\frac{\delta^2}{4\sigma}$ since any member of our finite set can be reconstructed  at this resolution. 
		\begin{small}
		\begin{align*}
			\E\Bigg[\max_{j = 1, \dots, M} \Big|\frac{1}{n}\sum_{i = 1}^{n} w_i\, g\,^j(\bx_i) \Big|\Bigg] &= \E\Bigg[\max_{j = 1, \dots, M} \frac{\Big|Z(g\,^j)\Big|}{\sqrt{n}}\Bigg]\\
			&\leq \frac{16}{\sqrt{n}} \int_{\frac{\delta^2}{4\sigma}}^{\delta} \sqrt{\log N_n(t; B_n(\delta; \cF))} dt
		\end{align*}
		\end{small}

		Hence $\cG_n(\delta) \leq \frac{\delta^2}{4\sigma} + \frac{16}{\sqrt{n}} \int_{\frac{\delta^2}{4\sigma}}^{\delta} \sqrt{\log N_n(t; B_n(\delta; \cF))} dt$. \qed
		\end{itemize}
	\end{frame}

	\section{Review and Examples}
	\frame{\sectionpage}
	
	\begin{frame}
		\frametitle{Review: Estimator, Models, and Definitions}
		\begin{block}{Model}
			Let $\cF$ be a function class and $\fstar \in \cF$ and $\{\bx_i\}_{i = 1}^{n}$ be a set of $n$ covariates. Assume that $y_i = \fstar(\bx_i) + \sigma w_i$ where $w_i \sim \mathcal{N}(0, 1)$.
		\end{block}
		\begin{block}{Estimator}
			Given the set $\{(\bx_1, y_1), \dots, (\bx_n, y_n) \}$:
			\begin{small}
			\vspace{-0.2cm}
			\begin{equation*}
				\fhat \in \arg\min_{f\in\cF} \frac{1}{n} \sum_{i = 1}^{n} \Big(y_i - f\,(\bx_i)\Big)^2.
			\end{equation*}
			\vspace{-0.3cm}
			\end{small}
		\end{block}
		\begin{block}{Measure of Goodness}
			Let $\{\bx_i\}_{i = 1}^{n}$ be the set of fixed covariates:
			\vspace{-0.3cm}
			\begin{small}
				\begin{equation*}
				||f-\fstar||_n := \Bigg[\frac{1}{n}\sum_{i = 1}^n \big(f\,(\bx_i) - \fstar(\bx_i)\big)^2\Bigg]^{1/2}.
				\end{equation*}
			\end{small}			
			\vspace{-0.3cm}
		\end{block}
	\end{frame}
	
	\begin{frame}				
		\begin{block}{Local Gaussian Complexity}
			\begin{equation*}
			\cG_n(\delta)  \E_w\Bigg[\sup_{
				\begin{scriptsize}
				\begin{matrix}
				g\in \cF\\||\,g\,||_n \leq \delta
				\end{matrix}
				\end{scriptsize}}\Big|\frac{1}{n}\sum_{i = 1}^{n}w_i\, f\;(\bx_i)\Big|\Bigg] 
			\end{equation*}
		\end{block}
		
		\begin{block}{Critical Inequality}
			A central object in our analysis is the set of positive scalars $\delta$ that satisfy the \textit{critical inequality}:
			\begin{equation*}
			\frac{\delta}{2\sigma} \geq \frac{\cG_n(\delta; \cF^\star)}{\delta}
			\end{equation*}			
			\vspace{-0.3cm}
		\end{block}										
		\begin{block}{Existence of Critical Radius}
			For any star shaped class $\mathcal{F}^\star$, the function $\delta \rightarrow \frac{\cG_n(\delta, \cF^\star)}{\delta}$ is non-increasing on $(0, \infty)$. Consequently, for any $c>0$, the inequality $\frac{\cG_n(\delta, \cF^\star)}{\delta} \leq c\delta$
			has a smallest positive solution.
		\end{block}			
	\end{frame}



	\begin{frame}
		\frametitle{Review: Main Results}
		\begin{block}{Theorem}
			Suppose that the shifted function class $\cF^\star$ is star-shaped, and let $\delta$ be any solution to the critical inequality. Then for any $t \geq \delta$, the nonparametric least-squares estimate $\widehat{f_n}$ satisfies the bound
			\begin{equation*}
			\Prob\big[||\widehat{f_n} - \fstar ||_n^2 \geq 16t\delta\big] \leq \exp\Big(\frac{-nt\delta}{2\sigma^2}\Big)
			\end{equation*}
		\end{block}
		\begin{block}{Auxiliary Lemma}
			Let $\cF^\star$ be a star-shaped class and $\delta$ satisfy the critical inequality.  Define the bad event $A(u)$ as
			\begin{small}
				
				\vspace{-0.3cm}
				\begin{equation*}
				A(u) = \Bigg\{\exists g \in \cF^\star, ||g||_n \geq u \;:\; \Big|\frac{\sigma}{n}\sum_{i = 1}^{n} w_i g(\bx_i)\Big|\geq 2u ||g||_n\Bigg\}.
				\end{equation*}
			\end{small}
			\vspace{-0.3cm}
			
			For a given $u \geq \delta$, we have 
			$\Prob[A(u)]\leq \exp\big(\frac{-nu^2}{2\sigma^2}\big)$.
		\end{block}
	\end{frame}	

	\begin{frame}
		\frametitle{Review: How to Compute a Critical Radius}
		For any star-shaped function class $\cF^\star$, define:
		\begin{itemize}
			\item $B\,(\delta\,; \cF^\star) = \{h\in \cF^\star \;\big|\; ||h||_n\leq \delta\}$
			\item $\mathcal{N}\,\big(t; B\,(\delta; \cF^\star)\big)$ be the $t$-convering number of $B\,(\delta; \cF)$ in norm $||.||_n$.
		\end{itemize}
		\begin{block}{Critical Inequality via Metric Entropy}
			Any $\delta \in (0, \infty]$ satisfying
			\begin{equation*}
			\frac{16}{\sqrt{n}} \int_{\frac{\delta^2}{4\sigma}}^{\delta} \sqrt{\log \mathcal{N}\big(t\,; B\,(\delta; \cF^\star)\big)} \;dt \leq \frac{\delta^2}{4\sigma},
			\end{equation*}
			satisfies the critical inequality.
		\end{block}
	\end{frame}

		\begin{frame}
		\frametitle{Parametric Example: Linear Regression}
		\begin{itemize}
			\item Consider $y_i = \langle \boldsymbol{\theta^\star}, \bx_i \rangle + w_i$ with $\boldsymbol{\theta}^\star \in \mathbb{R}^d$.
			\item Hypothesis class: $\cF = \{f_{\boldsymbol{\theta}}(.) = \langle \boldsymbol{\theta}, . \rangle\;|\; \boldsymbol{\theta} \in \mathbb{R}^d\}$.
			\item Let $\mathbf{X} \in \mathbb{R}^{d \times n}$ be the design matrix with rank $r$.
			\item $\cF = \cF^\star$ and convex. Thus star-shaped around any point.			
			\item $\forall  f_{\boldsymbol{\theta}} \in \cF$, the function $\boldsymbol{\theta} \to  ||f_{\boldsymbol{\theta}}\,||_n^2 = \frac{||\mathbf{X}\boldsymbol{\theta}||_2^2}{n}$.
			
			\item $B(\delta; \cF)$ is a $\delta$-ball in $\mathrm{Range}(\mathbf{X})$. 
			\item
			Dimension of $\mathrm{Range}(\mathbf{X})$ is $r$. By a volume argument:
			\begin{equation*}
			\log\Big(\mathcal{N}\big(t, B(\delta; \cF)\big)\Big) \leq r \log\big(1 + \frac{2\delta}{t}\big).
			\end{equation*}
			\item We have
			\begin{equation*}
			\frac{1}{\sqrt{n}} \int_{\frac{\delta^2}{4\sigma}}^{\delta} \sqrt{\log \mathcal{N}\big(t\,; B\,(\delta; \cF^\star)\big)} \;dt \leq c\delta \sqrt{\frac{r}{n}} \implies ||\fhat - \fstar||_n \leq c\sigma\sqrt{\frac{r}{n}}.
			\end{equation*}
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Non-Parametric Example: Lipschitz Functions}
		\begin{itemize}			
			\item Consider the function class 
			\begin{small}
				\begin{equation*}
					\cF_{\mathrm{Lip}}(L) = \{f:[0, 1] \to \mathbb{R} \big| f\,(0) = 0, f \text{ is L-Lip} \}.
				\end{equation*}			
			\end{small}
			\item Note that $\cF_{\mathrm{Lip}}(L) - \cF_{\mathrm{Lip}}(L) = 2 \cF_{\mathrm{Lip}}(L) \subseteq \cF_{\mathrm{Lip}}(2L)$.
			\item The Dudley's integral can be bounded as follows:
			\begin{small}
			\begin{align*}
				\frac{1}{\sqrt{n}}\int_{0}^{\delta}\sqrt{\log N_n\Big(t; B_n\big(\cF_{\mathrm{Lip}}(2L)\big)\Big)} \; dt\; &\lesssim\frac{1}{\sqrt{n}} \int_{0}^{\delta}\sqrt{\log N_\infty\Big(t; B_n\big(\cF_{\mathrm{Lip}}(2L)\big)\Big)} \; dt\\
				&\lesssim \frac{1}{\sqrt{n}}\int_{0}^{\delta} \Big(\frac{L}{t}\Big)^{\frac{1}{2}} \; dt \; \lesssim \sqrt{\frac{L\delta}{n}}.
			\end{align*}
			\end{small}	
			\item It suffices to choose $\delta>0$ such that $\sqrt{\frac{L\delta}{n}} \leq \frac{\delta^2}{\sigma} \implies \delta^2 \simeq \Big(\frac{L\sigma^2}{n}\Big)^{\frac{2}{3}}$.
			\item According to the main theorem: $||\fhat - \fstar||_n^2 \lesssim \Big(\frac{L\sigma^2}{n}\Big)^{\frac{2}{3}}$ with  probability at least $1 - c_1\exp\big(-c_2\big(\frac{n}{L\sigma^2}\big)^{\frac{1}{3}}\big)$.
		\end{itemize}
	\end{frame}

	\section{Oracle Inequalities}
	\frame{\sectionpage}
	
	\begin{frame}
		\frametitle{Oracle Inequalities}
		\begin{itemize}
			\item 		In our analysis so far, we have assumed that $\fstar \in \cF$. If we lift this assumption, we have:
			\begin{itemize}
				\item Prediction Error
				\item Approximation Error
			\end{itemize}
			\item It is natural to measure the approximation error in terms of the best approximating function in $\cF$, i.e. $\inf_{f\,\in \cF} || f - \fstar||_n$.
			\item Only an oracle that has access to $\fstar(\bx_i)$ can calculate $\inf_{f\,\in \cF} || f - \fstar||_n$. Hence we call inequalities with $\inf_{f\,\in \cF} || f - \fstar||_n$, the \textit{Oracle Inequalities}.
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Oracle Inequalities}
		Assume that $y_i = \fstar(\bx_i) + \sigma w_i$, where $w_i \sim \mathcal{N}(0, 1)$.\\ Define $\partial \cF = \{f_1 - f_2 | f_1, f_2 \in \cF\}$. We assume that this set is star-shaped. 	
		\begin{block}{Oracle Inequality}
			Let $\delta$ be any positive solution to the critical inequality
			$\frac{\cG_n(\delta, \partial \cF)}{\delta} \leq \frac{\delta}{2\sigma}.$

			For any $t\geq \delta$, the non-parametric least-square estimate $\fhat$ satisfies the bound
			\begin{small}
			\begin{equation*}
				||\fhat - \fstar||_n^2 \leq \inf_{\gamma \in (0, 1)}\Big\{\frac{1+\gamma}{1-\gamma} ||f-\fstar||_n^2 + \frac{c_0}{\gamma(1-\gamma)}t\delta\Big\} \text{  for all } f\in \cF, 
			\end{equation*}
			\end{small}
			with probability at least $1 - c_1 \exp(-\frac{c_2 n t\delta}{\sigma^2})$.			
			\end{block}		
		When $\fstar \in \cF$, we can set $f=\fstar$. Hence, $||\fhat - \fstar||_n^2 \lesssim t\delta$, recovering the previous result up to constants.		
	\end{frame}
	\begin{frame}
		\frametitle{Oracle Inequalities}
		\begin{equation*}
		||\fhat - \fstar||_n^2 \leq \inf_{\gamma \in (0, 1)}\Big\{\frac{1+\gamma}{1-\gamma} ||f-\fstar||_n^2 + \frac{c_0}{\gamma(1-\gamma)}t\delta\Big\} \text{  for all } f\in \cF. 
		\end{equation*}
		\begin{itemize}
			\item When $\fstar \notin \cF$, setting $t = \delta$ and taking infimum over $f\in\cF$, yields 
			\begin{equation*}
				||\fhat - \fstar||_n^2 \lesssim \inf_{f\,\in\cF}||f - \fstar||_n^2 + \delta^2
			\end{equation*}
			with high probability. The term $\delta^2$ can be viewed as the estimation error and $\inf_{f\,\in\cF}||\fhat - \fstar||_n^2$ as the approximation error.
		\end{itemize}
	\end{frame}


	\begin{frame}
	\frametitle{Oracle Inequalities: Proof}
		Given an arbitrary $\ftilde \in \cF$, define $\Dhat = \fhat - \fstar$ and $\Dtilde = \fhat - \ftilde$. We have 
		\begin{small}
		\begin{align*}
			0 &\geq \frac{1}{2n}\sum_{i = 1}^{n} \big(y_i - \fhat_i\,\big)^2 - \frac{1}{2n}\sum_{i = 1}^{n} \big(y_i - \ftilde_i\,\big)^2 \\
			&= \frac{1}{2}||\Dhat||_n^2 +  \frac{1}{2n}\sum_{i = 1}^{n} \big( y_i - {f_i\,^\star}\big)^2 +   \frac{1}{n}\sum_{i = 1}^{n} (\underbrace{y_i - f_i\,^\star}_{= \sigma w_i})(f_i\,^\star - \fhat_i\,)			
			 - \frac{1}{2n}\sum_{i = 1}^{n} \big(y_i - \ftilde_i\,\big)^2\\
		\end{align*}
		\end{small}
		Hence 
		\begin{small}
		$\frac{1}{2}||\Dhat||_n^2 \leq \frac{1}{2n}\sum_{i = 1}^{n} \big(\Dtilde_i - \Dhat_i\big) (2y_i - \ftilde_i - f_i\,^\star) - \frac{1}{n}\sum_{i = 1}^{n}\sigma w_i (\fstar_i - \fhat_i\,)$. 	
		\vspace{-0.5cm}
		
		\begin{align*}
			\frac{1}{2}||\Dhat||_n^2 &\leq \frac{1}{2n}\sum_{i = 1}^{n}\Big[ \big(\Dtilde_i - \Dhat_i\big) (2\sigma w_i - \Dtilde_i - \Dhat_i) + 2\sigma w_i \Dhat_i\Big]\\ 
			&=\frac{1}{n}\sum_{i = 1}^{n}\sigma w_i \Dtilde_i + \frac{1}{2} ||\fstar_i - \ftilde_i||_n^2	 \leq \frac{1}{2} \,||\fstar_i - \ftilde_i\,||_n^2 + \frac{\sigma}{n}\Big|\sum_{i = 1}^{n}w_i \Dtilde_i\Big|.
		\end{align*}
		\end{small}		
	\end{frame}
	
	\begin{frame}
	\frametitle{Oracle Inequalities: Proof}
	To bound the error, we consider two cases:
	\begin{itemize}
		\item \textbf{Case 1}: Suppose that $||\Dtilde||_n \leq \sqrt{t\delta}$.
			\begin{align*}
				||\Dhat||_n^2 &= ||(\ftilde - \fstar) + \Dtilde||_n^2\leq \Big(||\ftilde - \fstar||_n + \sqrt{t\delta}\Big)^2\\ &\leq (1+2\beta) ||\ftilde - \fstar||_n^2 + \big(1 + \frac{2}{\beta}\big)t\delta.				
			\end{align*}
			Set $\beta = \frac{\gamma}{1-\gamma}$. This yields 
			\begin{equation*}
			||\fhat - \fstar||_n^2 \leq \inf_{\gamma \in (0, 1)}\Big\{\frac{1+\gamma}{1-\gamma} ||\ftilde-\fstar||_n^2 + \frac{c_0}{\gamma(1-\gamma)}t\delta\Big\}.
			\end{equation*}
	\end{itemize}
	\end{frame}

	\begin{frame}
		\begin{itemize}
			\item \textbf{Case 2}: Suppose that $||\Dtilde||_n > \sqrt{t\delta}$.
			\begin{block}{Auxiliary Lemma}
				Let $\mathcal{H}$ be a star-shaped class and $\delta$ satisfy the critical inequality:
				\begin{small}
					
					\vspace{-0.3cm}
					\begin{equation*}
					\forall u\geq \delta:\;\;\;\Prob\Bigg[\exists\, g \in \mathcal{H}, ||g||_n \geq u \;:\; \Big|\frac{\sigma}{n}\sum_{i = 1}^{n} w_i g(\bx_i)\Big|\geq 2u ||g||_n\Bigg] \leq \exp\big(\frac{-nu^2}{2\sigma^2}\big).
					\end{equation*}
				\end{small}
				\vspace{-0.3cm}
				
			\end{block}
			Let $u = \sqrt{t\delta}$ and $\mathcal{H} = \partial \cF$. We have
			\begin{small}
			\begin{equation*}
				\Prob\Big[2\big|\frac{\sigma}{n} \sum_{i = 1}^{n}w_i \Dtilde(\bx_i)\big|\geq 4\sqrt{t\delta}||\Dtilde||_n\Big]\leq \exp\big(-\frac{nt\delta}{2\sigma^2}\big).
			\end{equation*}
			\end{small}
			\vspace{-0.3cm}
			
			Hence, with high probability
			\begin{small}
			\begin{align*}
				||\Dhat||_n^2 &\leq ||\ftilde - \fstar||_n^2 + 4\sqrt{t\delta} ||\Dtilde||_n^2
				\leq ||\ftilde - \fstar||_n^2 + 4\sqrt{t\delta} \,\Big\{||\Dhat||_n^2 + ||\ftilde - \fstar||_n^2\Big\}\\
				&\leq ||\ftilde - \fstar||_n^2 + \Big[4\beta ||\Dhat||_n^2 + \frac{4}{\beta}t \delta\Big] + \Big[4\beta||\ftilde - \fstar||_n^2 + \frac{4}{\beta} t\delta\Big].
			\end{align*}
			\end{small}
			Rearrenging terms: $
				||\Dhat||_n^2 \leq \frac{1+4\beta}{1 - 4\beta} ||\ftilde - \fstar||_n^2 + \frac{8}{\beta(1-4\beta)}t\delta.$ \qed
		\end{itemize}				
	\end{frame}

			\item Viewed as a function of $\mathbf{w}$, $Z_n(S)$ is $\frac{\delta}{\sqrt{n}}$ Lipschitz. Hence
			\begin{small}

					\Prob\big[Z_n(S) \geq \E(Z_n(S)) + t\delta\big] \leq \exp\Big(\frac{-nt^2}{2}\Big)

			\end{small}						
		\end{itemize}

	\begin{frame}
		\begin{itemize}
			\item Consider the SVD of $\mathbf{X}_S = \mathbf{U}\mathbf{D}\mathbf{V}^\top$. We have $||\mathbf{X}_S\boldsymbol{\theta}_S||_2 = ||\mathbf{D}\mathbf{V}^\top\boldsymbol{\theta}_S||_2$:
			\begin{small}
				\begin{equation*}
					\E\big[Z_n(S)\big] = \E \Big[\sup_{
					\begin{scriptsize}
					\begin{matrix}
					\beta \in \mathbb{R}^{2s}\\
					||\beta||_2 \leq \delta
					\end{matrix}					
					\end{scriptsize}

				\end{equation*}
			\end{small}
		\item Since $\mathbf{U}$ is orthogonal and $\mathbf{w}\sim \mathcal{N}(0, \mathbf{I}_{n})$, thus $\mathbf{U}^\top\mathbf{w}\sim \mathcal{N}(0, \mathbf{I}_{2s})$.

		\begin{small}

			\end{equation*}
		\end{small}
		\item By integrating:
		\begin{small}
			\begin{equation*}
				\frac{\E[\max_{|S|=2s} Z_n(S)]}{\delta} = \frac{\cG_n(\delta)}{\delta} \lesssim \sqrt{\frac{s}{n}} + \sqrt{\frac{\log{d \choose 2s}}{n}} \lesssim \sqrt{\frac{s\log\big(\frac{ed}{s}\big)}{n}}.
			\end{equation*}
		\end{small}
		\item Thus $\delta \simeq \sigma^2 \frac{s\log(ed/s)}{n}$ satisfies critical inequality.
		\end{itemize}

	\section{Regularized Estimators}
	\frame{\sectionpage}
	
	\begin{frame}
		\frametitle{Oracle Inequalities for Regularized Estimators}
		\begin{itemize}
			\item Given a space $\cF$ of real-valued functions, an associated norm $||.||_\cF$, consider the family of regularized least-square problems:
			\begin{equation*}
				\fhat \in \arg\min_{\cF} \Big\{\frac{1}{2n} \sum_{i = 1}^{n} \big(y_i - f\,(\bx_i)\big)^2 + \lambda_n ||f\,||_\cF^2\Big\}.
			\end{equation*}
			\item A Local Gaussian Complexity Measure:
			\begin{equation*}
				\cG_n(\delta) \triangleq \cG_n\big(\delta; B_{\partial\cF}(3)\big) = \E_w\Bigg[\sup_{
					\begin{scriptsize}
					\begin{matrix}
					g\in \partial\cF\\||\,g\,||_n \leq \delta\\ ||\,g\,||_\cF \leq 3
					\end{matrix}
					\end{scriptsize}}\Big|\frac{1}{n}\sum_{i = 1}^{n}w_i\, f\;(\bx_i)\Big|\Bigg] 
			\end{equation*}
			\item The critical inequality for a user defined $R$:
			\begin{equation*}
				\frac{\cG_n(\delta)}{\delta}\leq \frac{R}{2\sigma}\delta.
			\end{equation*}
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Oracle Inequalities for Regularized Estimators}
		\begin{block}{Theorem}
			Consider a convex function class $\cF$ and assume that $\delta$ satisfies the critical inequality
			\begin{equation*}
			\frac{\cG_n(\delta)}{\delta}\leq \frac{R}{2\sigma}\delta.
			\end{equation*}
			 If $\lambda_n \geq 2\delta^2$. There exists universal constants $c_0, c_1, c_2, c_3$ such that 
			\begin{equation*}
			||\fhat - \fstar||_n^2 \leq c_0 \inf_{||f\,||_\cF \leq R} ||f - \fstar||_n^2 + c_1 R^2(\delta^2 +\lambda_n)

\begin{frame}
	\frametitle{Local Gaussian Complexity of unit Ball of an RKHS}
	\begin{block}{Theorem}
		Consider an RKHS with kernel $k$. For a given set of points $\{\bx_i\}_{i = 1}^{n}$, let $\hat{\mu}_1 \geq \hat{\mu}_2 \geq \dots \geq \hat{\mu}_n \geq 0$ be the eigenvalues of normalized kernel matrix $\mathbf{K}$ with entries $\mathbf{K}_{ij} = k(\bx_i, \bx_j)/n$. For all $\delta >0$, we have
		\begin{equation*}
			\E\Bigg[\sup_{
		\begin{scriptsize}
		\begin{matrix}
		||f\,||_\cF \leq 1\\
		||f\,||_n\leq \delta
		\end{matrix}
		\end{scriptsize}}
	\Big|\frac{1}{n} \sum_{i = 1}^{n} w_i f\,(\bx_i)\,\Big|\Bigg] \leq \sqrt{\frac{2}{n}}\sqrt{\sum_{j = 1}^{n} \min\big(\delta^2, \hat\mu_j\big)}
		\end{equation*}
	\end{block}
		It suffices to consider functions of the form $g(.) = \frac{1}{\sqrt{n}} \sum_{i = 1}^{n} \alpha_i k(., \bx_i)$. Any function of the Hilbert space $\cF$ can be written as $f = g + g^\perp$. We must have $g\,^\perp(\bx_i) = \langle g\,^\perp, k\,(., \bx)\rangle = 0$. We also have $||f\,||_\cF^2 = ||g\,||_\cF^2 + ||g\,^\perp||_\cF^2$. Without loss of generality, we can assume $g\,^\perp = 0$.	
\end{frame}
\begin{frame}
	\begin{itemize}
		\item The constraint $||g\,||_n\leq \delta$ is equivalent to $||\mathbf{K}\alpha||_2 \leq \delta$.
		\item The constraint $||g\,||_\cF\leq 1$ is equivalent to $\alpha^\top\mathbf{K}\alpha \leq 1$.
		\item The local complexity:
		\begin{equation*}
		\cG_n(\delta) = \frac{1}{\sqrt{n}}\E\Bigg[\sup_{
			\begin{scriptsize}
			\begin{matrix}
			\alpha^\top \mathbf{K} \alpha \leq 1\\
			\alpha \mathbf{K}^2\alpha \leq \delta^2
			\end{matrix}
			\end{scriptsize}}
		\Big|w^\top \mathbf{K} \alpha\Big|\Bigg]
		\end{equation*}
		\item $\mathbf{K}$ is PSD, hence $\mathbf{K} = \mathbf{U}^\top \Lambda \mathbf{U}$: $
		\cG_n(\delta) = \frac{1}{\sqrt{n}} \E\big[\sup_{\beta \in D} |w^\top \beta|\big]$
		where $D = \Big\{\beta\in \mathbb{R}^n\; \big| \;||\beta\,||_2^2 \leq \delta^2, \sum_{j = 1}^{n}\frac{\beta_j^2}{\hat \mu_j} \leq 1\Big\}$
		\item Define $\mathcal{E} = \{\beta \in \mathbb{R}^d\; \big|\; \sum_{j = 1}^{n} \eta_j \beta_j^2 \leq 2\}$, where $\eta_j = \max\{\delta^{-2}, \hat{\mu}_j^{-1}\}$
		\item For any $\beta \in D$, we have $
			\sum_{j = 1}^{n} \max\{\delta^{-2}, \hat{\mu}_j^{-1}\} \beta_j^2 \leq \sum_{j = 1}^{m} \frac{\beta_j^2}{\delta^2} + \frac{\beta_j^2}{\hat\mu_j} \leq 2.$ Hence $D \subseteq D$.
		\item As a result, by H\"older Inequality
		\begin{equation*}
			\cG_n(\delta) \leq \sqrt{\frac{2}{n}}\E\sqrt{\sum_{j = 1}^{n} \frac{w_j^2}{\eta_j}} \leq \sqrt{\frac{2}{n}}\sqrt{\sum_{j = 1}^{n} \frac{1}{\eta_j}}. \qed
		\end{equation*}
	\end{itemize}
\end{frame}
