\section{ماتریس‌های زیرگوسی}
فرض کنید که ماتریس های 
$\{Q_i\}_{i=1}^{n}$
،
ماتریس های متقارن
$d$
بعدی باشند که 
$Q_i$
زیرگوسی با پارامتر
$V_i$
باشند.
تعریف می‌کنیم:
$$S_n= \frac{1}{n} \sum_{i=1}^{n} Q_i$$
\begin{enumerate}
    \item 
    نشان دهید که:
    $$\mathbb{E}(\gamma_{max}(S_n)) \leq \sqrt{\frac{2 \sigma^2 \log{d}}{n}}, \ \ \text{\lr{where}} \ \ \sigma^2= ||\frac{1}{n} \sum_{i=1}^{n} V_i||_2 $$
    که در آن 
    $\gamma_{max}(S_n)$
    بزرگترین مقدار ویژه است.
    \item
    نشان دهید که:
    $$\mathbb{E}(||S_n||_2) \leq \sqrt{\frac{2 \sigma^2 \log{2d}}{n}}$$
\end{enumerate}
